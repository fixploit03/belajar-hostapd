# Konfigurasi Hostapd OPN

```bash
# Konfigurasi Hostapd - Open Network (OPN)

# Nama interface wireless
interface=wlan0
# Nama driver Wi-Fi
driver=nl80211
# Nama Wi-Fi
ssid=Wi-Fi OPN
# Mode Wi-Fi
hw_mode=g
# Channel Wi-Fi
channel=6
# Kode Negara
country_code=ID

# -------- Open Network --------

auth_algs=1     # 1 = Open System Authentication
wpa=0           # 0 = Tidak menggunakan WPA/WPA2
```
